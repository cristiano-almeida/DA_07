📌 Classe: DesafioConta.java

📝 Código para completar:

package tema07;

public class DesafioConta {
    // ATRIBUTOS PRIVADOS
    private String titular;
    private int numero;
    private  saldo;

    // ========== GETTERS ==========
    public String () {
        return titular;
    }

    public int () {
        return numero;
    }

    public double () {
        return saldo;
    }

    // ========== SETTERS ==========
    public void (String titular) {
        if (titular != null && !titular.trim().isEmpty()) {
            this.titular = titular;
        } else {
            System.out.println("❌ Titular inválido!");
        }
    }

    public void (int numero) {
        if (numero > 0) {
            this.numero = numero;
        } else {
            System.out.println("❌ Número da conta inválido!");
        }
    }

    // ========== MÉTODOS ==========
    public void (double valor) {
        if (valor > 0) {
            saldo += valor;
            System.out.println("Depósito de R$ " + valor + " realizado!");
            System.out.println("Saldo atual: R$ " + saldo);
        } else {
            System.out.println("❌ Valor de depósito inválido!");
        }
    }

    public void (double valor) {
        if (valor > 0 && valor <= saldo) {
            saldo -= valor;
            System.out.println("Saque de R$ " + valor + " realizado!");
            System.out.println("Saldo atual: R$ " + saldo);
        } else if (valor <= 0) {
            System.out.println("❌ Valor de saque inválido!");
        } else {
            System.out.println("❌ Saldo insuficiente!");
        }
    }

    public void () {
        System.out.println("Titular: " + titular);
        System.out.println("Número: " + numero);
        System.out.println("Saldo: R$ " + saldo);
    }

    // ========== MAIN PARA TESTE ==========
    public static void main(String[] args) {
        DesafioConta conta = new DesafioConta();

        System.out.println("--- TESTE: Operações válidas ---");
        conta.("Maria Santos");
        conta.(12345);
        conta.();

        System.out.println("\n--- DEPOSITANDO ---");
        conta.(1000);

        System.out.println("\n--- SACANDO ---");
        conta.(300);

        System.out.println("\n--- TESTE: Operações INVÁLIDAS ---");
        conta.(-50);      // ❌ Bloqueado!
        conta.(1000);     // ❌ Saldo insuficiente!
        conta.();
    }
}
